    Python

    With this functionality you can run Python scripts in BricsCAD BIM.

    Running Python

    1. Enter the BIMPYTHON command in the command bar to initialize the Python process. The Python built-ins and BriqPy are loaded.
    2. Select the Python script to run.

    Usage

    Example to get some info about the lengths of the walls in the drawing:
    lengths = []

for wall in bim_model.filter(Type='Wall'):
    lengths.append(wall.prop('Length'))

print(f'wall lengths. max: {max(lengths)}, avg: {sum(lengths)/len(lengths)}')
    Example to create a selection and print the objects:
    bim_model.filter(Type='Wall', IsExternal=True).select()

    for wall in bim_model.filter(Type='Wall', IsExternal=True, Length=max(lengths)):
    print(wall)
    Example to get the parts of the roof that are close to a wall:
    roof_parts = bim_model.filter(Type='Roof').parts()

roof_parts_close_to_wall = bim_model.filter(Type='Wall').within_distance(40, 'cm', search_range=roof_parts)
    Example to filter with functions:
    def is_long(obj): return obj.prop('Length') > 50

roof_parts.filter(is_long)
    Example to export data:
    # create a dictionary list
wall_info = [
    {
        'Handle': wall.prop('Handle'),
        'Length': wall.prop('Length'),
        'Height': wall.prop('Height')
    } for wall in bim_model.filter(Type='Wall')]

# export to .json
import json
file = open('path/to/file.json', 'w+')
file.write(json.dumps(wall_info, indent=4))
file.close()

# plotting a histogram
import matplotlib.pyplot as plt
import pandas as pd
df = pd.DataFrame(wall_info)
df.hist();
plt.show()

# export to .csv
df.to_csv(r'path/to/file.csv', index = False, header=True)
    API
    # class representing BIM objects
class briqpy.Object
    # get property value
    def prop(propName)

    # calculate the distance between two objects
    # options for unit: any unit form insunits ('Centimeters', 'Feet', 'Parsecs' etc.) and these abbreviations: 'mm', 'cm, 'm', 'km', 'ft'
    def distance_to(otherObj, unit='undefined')

    # get the sub-elements of this object
    # to return non empty range, this object has to be a block reference containing bim objects
    def parts()

    # get the parent element of this sub-element
    def parent()

    # get a list of objects that are within the clearance of arg distance
    # options for distance_mode: 'bbox_center', 'bbox', exact'
    # options for unit: see distance_to
    # it's necessary to provide a search_range for nested objects (coming form parts())
    def within_distance(distance, unit='undefined', distance_mode='exact', search_range=bim_model)

    # add object to selection
    def select()

    # remove object from selection
    def deselect()
    # class representing a collection BIM objects
class briqpy.ObjectRange
    # filtering the range with a function
    def filter(function)

    # filtering the range with conditions given as keyword arguments
    def filter(**conditions)

    # get a flat list of all the parts of all the elements in this range
    def parts()

    # get all parent objects of elements in this range 
    def parents()

    # get a flat list of objects that are within distance to any object in this range
    def within_distance(distance, unit='undefined', distance_mode='exact', search_range=bim_model)

    # add objects to selection
    def select()

    # remove objects from selection
    def deselect()

    # number of elements in the object range
    def count()
